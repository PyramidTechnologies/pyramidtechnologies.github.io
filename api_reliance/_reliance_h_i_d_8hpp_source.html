8  * RelianceCore API is a native C++ library for monitoring your Reliance Thermal Printer.
9  * This library support status checks and a number of configuration options.
10  *
11  * Getting started requires 3 files:
12  *
13  * - RelianceHID.dll
14  * - RelianceHID.lib
15  * - RelianceHID.hpp
16  *
17  * 1. Place the DLL where your app can find it at run time
18  * 2. Link the library during compilation time
19  * 3. Include the header in your project
20  *
21  * Sample usage:
22  * @code
23  * ...
24  * #include <iostream>
25  * #include <RelianceHID.hpp>
26  *
27  * // Use namespace for brevity
28  * using namespace reliancecore;
29  * ...
30  *
31  * // Create an initialize your printer
32  * RelianceHID printer;
33  * auto result = printer.initialize();
34  *
35  * // result holds the initialization success/error messages
36  * std::cout << result.message.c_str() << std::endl;
37  *
38  * // Check the status
39  * RelianceHID::RelianceStatus status;
40  * result = printer.getStatus(&status);
41  *
42  * // result holds the command success/error messages
43  * std::cout << result.message.c_str() << std::endl;
44  *
45  * if(status.ret == RelianceHID::ReturnCode::Okay)
46  * {
47  * // explore the status struct, this contains complete status
48  * ...
49  * }
50  *
51  * @endcode
52  *
53  * Please contact <mailto:support@pyramidacceptors.com> to request the right DLL version
54  * for your system.
55  *
56  */
