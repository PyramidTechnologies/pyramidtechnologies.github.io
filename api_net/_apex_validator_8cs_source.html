303  private void ReadAcceptorResp()
304  {
305  var data = PreviouslySentMasterMsg ?? GenerateNormalMessage();
306 
307  // Attempt to write data to slave
308  NotifySerialData(DebugBufferEntry.AsMaster(data));
309  WriteWrapper(data);
310 
311 
312  // Blocks until all 11 bytes are read or we give up
313  var resp = ReadWrapper();
314 
315 
316  // Extract only the states and events
317  NotifySerialData(DebugBufferEntry.AsSlave(resp));
318 
319  // No data was read, return!!
320  if (resp.Length == 0)
321  {
322  // Do no toggle the ack
323  return;
324  }
325 
326  // Check that we have the same ACK #
327 
328  if (IsBadAckNumber(resp, data))
329  {
330  PreviouslySentMasterMsg = data;
331  return;
332  }
333 
334  if (IsBadSlaveChecksumOk(resp))
335  {
336  PreviouslySentMasterMsg = data;
337  return;
338  }
339 
340  // Otherwise we're all good - toggle ack and clear last sent message
341 
342  PreviouslySentMasterMsg = null;
343  Ack ^= 1;
344 
345  // At this point we have sent our message and received a valid response.
346  // Parse the response and act on any state or events that are reported.
347 
348 
349  // Translate raw bytes into friendly enums
350  var slaveMessage = SlaveCodex.ToSlaveMessage(resp);
351  PreviousState = SlaveCodex.GetState(slaveMessage);
352 
353 
354  // Raise a state changed notice for clients
355  NotifyStateChange(PreviousState);
356 
357 
358  // Multiple event may be reported at once
359  var currentEvents = SlaveCodex.GetEvents(slaveMessage);
360  foreach (Events e in Enum.GetValues(typeof(Events)))
361  {
362  // If flag is set in slave message, report event
363  if ((currentEvents & e) == e)
364  {
365  NotifyEvent(e);
366  }
367  }
368 
369  // Check for cassette missing - reports every time cashbox is missing
370  if (!SlaveCodex.IsCashboxPresent(slaveMessage))
371  {
372  CashboxPresent = false;
373 
374  NotifyError(Errors.CashboxMissing);
375  }
376  // Only report the cashbox attached 1 time after it is re-attached
377  else if (!CashboxPresent)
378  {
379  CashboxPresent = true;
380 
381  SafeEvent(OnCashboxAttached);
382  }
383 
384 
385  // Mask away rest of message to see if a note is in escrow. If this is the first
386  // escrow message, start the escrow timeout clock
387  if (!NoteIsEscrowed && PreviousState == States.Escrowed)
388  {
389  _escrowStart = DateTime.MinValue;
390  }
391 
392  NoteIsEscrowed = PreviousState == States.Escrowed;
393 
394  // Credit bits are 3-5 of data byte 3
395  var value = SlaveCodex.GetCredit(slaveMessage);
396  if (value > 0)
397  {
398  Credit = (byte) value;
399  }
400 
401  // Per the spec, credit message is issued by master after stack event is
402  // sent by the slave. If the previous event was stacked or returned, we
403  // must clear the escrow command to completely release the note from
404  // escrow state.
405  // ReSharper disable once SwitchStatementMissingSomeEnumCasesNoDefault
406  switch (PreviousEvents)
407  {
408  case Events.Stacked:
409  NotifyCredit(Credit);
410  // C# does not allow fallthrough so we will goto :)
411  goto case Events.Returned;
412 
413  case Events.Returned:
414  // Clear our the pending escrow command once we've stacked or returned the note
415  EscrowCommand = EscrowCommands.None;
416  Credit = 0;
417  break;
418  }
419 
420  // Update the events aster the check for check so as to not lose a credit message
421  PreviousEvents = currentEvents;
422  }

429  private byte[] GenerateNormalMessage()
430  {
431  // # basic message 0 1 2 3 4 5 6 7
432  // start, len, ack, bills,escrow,resv'd,end, checksum
433  var data = Request.BaseMessage;
434 
435  // Toggle message number (ack #) if last message was okay and not a re-send request.
436  data[2] = (byte) (0x10 | Ack);
437 
438  // Get enable mask from client configuration. On next message, the acceptor
439  // will update itself and not escrow any notes that are disabled in this mask.
440  data[3] = Config.EnableMask;
441 
442 
443  if (!Config.IsEscrowMode)
444  {
445  // Clear escrow mode bit
446  data[4] = 0x00;
447 
448  if (NoteIsEscrowed)
449  {
450  data[4] |= 0x20;
451  }
452  }
453  else
454  {
455  // Set escrow mode bit
456  data[4] = 1 << 4;
457 
458  if (!NoteIsEscrowed)
459  {
460  return Checksum(data);
461  }
462 
463  // Note is in escrow, we have not yet notified the client, and we did not just stack
464  // a note. Due to the polling loop, we will always send an escrow message immediately
465  // after a stack because we are really reporting on state during the prior poll.
466  if (EscrowCommand == EscrowCommands.None)
467  {
468  EscrowCommand = EscrowCommands.Notify;
469  }
470 
471  // Otherwise do what the host tells us to do.
472  // ReSharper disable once SwitchStatementHandlesSomeKnownEnumValuesWithDefault
473  switch (EscrowCommand)
474  {
475  case EscrowCommands.Stack:
476  // set stack bit
477  data[4] |= 0x20;
478  EscrowCommand = EscrowCommands.Acknowledged;
479  break;
480 
481  case EscrowCommands.Reject:
482  // set reject bit
483  data[4] |= 0x40;
484  EscrowCommand = EscrowCommands.Acknowledged;
485  break;
486 
487  case EscrowCommands.Notify:
488  // notify client we have a bill in escrow
489  NotifyEscrow(Credit);
490  EscrowCommand = EscrowCommands.Awaiting;
491  break;
492 
493  case EscrowCommands.Awaiting:
494  // Perform timeout check and set reject flag is timeout exceeded if we are awaiting stack/return command
495  if (Config.EscrowTimeoutSeconds > 0)
496  {
497  if (_escrowStart == DateTime.MinValue)
498  {
499  _escrowStart = DateTime.Now;
500  }
501 
502  var delta = DateTime.Now - _escrowStart;
503  if (delta.TotalSeconds > Config.EscrowTimeoutSeconds)
504  {
505  EscrowCommand = EscrowCommands.Reject;
506 
507  _escrowStart = DateTime.MinValue;
508  }
509  }
510 
511  break;
512 
513  case EscrowCommands.Acknowledged:
514  EscrowCommand = EscrowCommands.None;
515  break;
516 
517  case EscrowCommands.None:
518  // do nothing
519  break;
520  }
521  }
522 
523 
524  // Set the checksum
525  return Checksum(data);
526  }

603  private static byte[] Checksum(IList<byte> msg)
604  {
605  var tmp = new List<byte>(msg);
606  var checksum = (byte) (msg[1] ^ msg[2]);
607  for (var i = 3; i < msg.Count - 1; i++)
608  {
609  checksum ^= msg[i];
610  }
611 
612  tmp.Add(checksum);
613  return tmp.ToArray();
614  }

627  private static bool IsBadSlaveChecksumOk(byte[] msg)
628  {
629  // If the length is incorrect, the checksum doesn't matter
630  if (msg.Length != 11)
631  {
632  return true;
633  }
634 
635  // msg should be the entire message, including checksum. -2 to skip checksum
636  var checksum = (byte) (msg[1] ^ msg[2]);
637  for (var i = 3; i < msg.Length - 2; i++)
638  {
639  checksum ^= msg[i];
640  }
641 
642  return checksum != msg[10];
643  }
